  1. gtid简介 gtid，全局事务标识，具有全局唯一且具有幂等性

何为幂等性，所谓幂等性，就是无论执行多少次，结果是一样的

2 gtid的组成 gtid=server_uuid:transaction_id server_uuid，保存在mysql数据目录下的auto.conf，可通过cat命令查看

[root@localhost ~]# cat /liwuming/mysql/auto.cnf
[auto]
server-uuid=479ff3ca-1a1d-11ed-830e-000c2904423a

在mysql开机时会检测该文件，如果不存在会生成新的auto.conf

transaction_id,当前服务器提交事务的序列号，从1开始自增长，一个数值对应一个事务

开启gtid

[mysqld]
xxx

gtid_mode=on
enforce_gtid_consistency=on
#防止复制随着mysql启动而自动启动
skip_slave_start=1

刷新binlog flush logs;

重置binlog，会删除所有binlog，从0开始

reset master;

--include-gtids=name

--exclude-gtids=name

mysqlbinlog --exclude-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:1-4,6 mysql-binlog.000001

--include-gtids与--exclude-gtids可以一起搭配使用且可以使用多次，这样就使得截取更加灵活

mysqlbinlog --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:1 --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:2 --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:3 --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:4 --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:5 --include-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:6 --exclude-gtids=479ff3ca-1a1d-11ed-830e-000c2904423a:5 mysql-binlog.000001

mysqlbinlog --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:1' --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:2' --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:3' --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:4' --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:5' --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:6' --exclude-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:5' mysql-binlog.000001

mysqlbinlog --skip-gtids --include-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:1-6' --exclude-gtids='479ff3ca-1a1d-11ed-830e-000c2904423a:5' mysql-binlog.000001>/liwuming/backup1.sql
